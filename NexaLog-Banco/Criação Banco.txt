CREATE DATABASE NexaLog;
GO
USE NexaLog;
GO

CREATE TABLE Usuario (
    idusuario INTEGER IDENTITY PRIMARY KEY,
    email VARCHAR(100),
    nome VARCHAR(100),
    senha VARCHAR(255),
    tipousuario VARCHAR(20)
);

CREATE TABLE Produto (
    idproduto INTEGER IDENTITY PRIMARY KEY,
    codProduto INTEGER,
    nome VARCHAR(100),
    dataCadastro DATE,
    dataValidade DATE,
    quantidade DECIMAL(10,2),
    unidadeMedida VARCHAR(5) NOT NULL DEFAULT 'Kg',
    descricao VARCHAR(255)
);

CREATE TABLE Movimentacao (
    id_movimentacao INTEGER IDENTITY PRIMARY KEY,
    tipo_movimentacao VARCHAR(20),
    data_movimentacao DATE,
    fk_Usuario_idusuario INTEGER
);

CREATE TABLE Lote (
    quantidade_lote DECIMAL(10,2),,
    id_lote INTEGER  IDENTITY PRIMARY KEY,
    cod_lote VARCHAR(50),
    dataValidade DATE,
    dataFabricacao DATE,
    fk_Produto_idproduto INTEGER
);

CREATE TABLE Item_Movimentacao (
    fk_Produto_idproduto INTEGER,
    fk_Movimentacao_id_movimentacao INTEGER,
    quantidade DECIMAL(10,2)
);
 
ALTER TABLE Movimentacao ADD CONSTRAINT FK_Movimentacao_2
    FOREIGN KEY (fk_Usuario_idusuario)
    REFERENCES Usuario (idusuario)
    ON DELETE NO ACTION;
 
ALTER TABLE Lote ADD CONSTRAINT FK_Lote_2
    FOREIGN KEY (fk_Produto_idproduto)
    REFERENCES Produto (idproduto)
    ON DELETE CASCADE;
 
ALTER TABLE Item_Movimentacao ADD CONSTRAINT FK_Item_Movimentacao_1
    FOREIGN KEY (fk_Produto_idproduto)
    REFERENCES Produto (idproduto)
    ON DELETE NO ACTION;
 
ALTER TABLE Item_Movimentacao ADD CONSTRAINT FK_Item_Movimentacao_2
    FOREIGN KEY (fk_Movimentacao_id_movimentacao)
    REFERENCES Movimentacao (id_movimentacao)
    ON DELETE CASCADE;
